// ── Classification ──
function cls(name, operator, country) {
  const n=(name||'').toLowerCase();const op=(operator||'').toLowerCase();
  // Non-US: NHS/HSE/SNS countries — mostly public by default
  if(country!=='United States'&&country!=='Australia'){
    if(op.includes('nhs')||op.includes('health service')||op.includes('hse')||op.includes('health board')||op.includes('trust')||op.includes('hsc')||op.includes('foundation trust')||op.includes('national health')||op.includes('servicio de salud')||op.includes('sistema nacional')||op.includes('ministry')||op.includes('government')||op.includes('department of health')||op.includes('army')||op.includes('navy')||op.includes('air force')||op.includes('military'))return'public';
    if(n.includes('nhs '))return'public';
    const iePub=['beaumont','st. james','mater','st. vincent','tallaght','cork university','galway university','connolly','mercy university','letterkenny','sligo university','waterford university','portiuncula','mayo university','naas general','wexford general','cavan general','louth county','our lady of lourdes','st. luke','bantry','mallow','south infirmary','tipperary','midland regional','national maternity','coombe','rotunda','children\'s health ireland','temple street','crumlin'];
    if(country==='Ireland'&&iePub.some(k=>n.includes(k)))return'public';
    if(country==='Spain')return'public';
    if(country==='United Kingdom'&&(n.startsWith('royal ')||n.includes('royal hospital')||n.includes('royal infirmary')||n.includes('university hospital')||n.includes('teaching hospital')||n.includes('general hospital')||n.includes('district')))return'public';
  }
  // Private operators (any country)
  if(op.includes('private')||op.includes('ltd')||op.includes('limited')||op.includes('plc')||op.includes('inc.')||op.includes('corporation')||FOR_PROFIT_OPS.some(f=>op.includes(f))||op==='nuffield health'||op==='spire healthcare'||op==='circle health'||op==='bmi healthcare'||op==='ramsay health care'||op==='quiron'||op==='hospitales universitarios')return'private';
  if(n.includes('private')||n.includes('clínica'))return'private';
  if(country==='United Kingdom'&&(n.includes('nuffield')||n.includes('spire ')||n.includes('bmi')||n.includes('ramsay')||n.includes('circle health')))return'private';
  // USA
  if(country==='United States'){
    if(US_PUB_NAME.some(k=>n.includes(k)))return'public';
    if(op.includes('veterans')||op.includes('county')||op.includes('state')||op.includes('federal')||op.includes('government')||op.includes('army')||op.includes('navy')||op.includes('air force')||op.includes('united states')||op.includes('prison')||op.includes('correction')||op.includes('nyc health')||op.includes('department of')||op.includes('public health'))return'public';
    if(US_FP_NAME.some(k=>n.includes(k)))return'private';
    if(US_FP_OP.some(f=>op.includes(f)))return'private';
    if(US_NP_NAME.some(k=>n.includes(k)))return'nonprofit';
    return'nonprofit';
  }
  // Australia
  if(country==='Australia'){
    if(n.startsWith('royal')||n.includes('hospital')||n.includes('health service')||n.includes('prince')||n.startsWith('st '))return'public';
    if(n.includes('private'))return'private';return'public';
  }
  if(country==='United Kingdom'){if(n.includes('private'))return'private';return'public';}
  return'unknown';
}

// ── Fetch ──
function fetchH(){
  const cl=COUNTRIES.join(', ');
  const q=`SELECT ?h ?hLabel ?coord ?opLabel ?cLabel ?beds WHERE {
  ?h wdt:P31 wd:Q16917; wdt:P625 ?coord; wdt:P17 ?c.
  FILTER(?c IN (${cl})) OPTIONAL { ?h wdt:P137 ?op. } OPTIONAL { ?h wdt:P1313 ?beds. }
  SERVICE wikibase:label { bd:serviceParam wikibase:language "en". } } LIMIT 10000`;
  document.getElementById('load-progress').textContent='Querying Wikidata...';
  fetch(WD+'?format=json&query='+encodeURIComponent(q),{headers:{'Accept':'application/json','User-Agent':'HM/1.0'}})
    .then(r=>r.json()).then(d=>{
      const parsed=d.results.bindings.map(b=>{
        const m=b.coord.value.match(/Point\(([\d.-]+)\s+([\d.-]+)\)/);if(!m)return null;
        const [lon,lat]=[parseFloat(m[1]),parseFloat(m[2])];if(isNaN(lat)||isNaN(lon))return null;
        const name=b.hLabel.value,op=b.opLabel?b.opLabel.value:'',ctry=b.cLabel.value;
        return{n:name,lat,lng:lon,t:cls(name,op,ctry),m:mental(name),op,b:b.beds?parseInt(b.beds.value):0,ctry};
      }).filter(Boolean);
      allH=parsed;
      const byC={};parsed.forEach(h=>{byC[h.ctry]=(byC[h.ctry]||0)+1;});
      // Count types per US
      const usT={public:0,private:0,nonprofit:0};parsed.filter(h=>h.ctry==='United States').forEach(h=>{usT[h.t]++;});
      document.getElementById('load-progress').textContent=Object.entries(byC).map(([c,n])=>`${FLAGS[c]||c} ${n}`).join(' · ')+` | US: Pub ${usT.public} Pri ${usT.private} NP ${usT.nonprofit}`;
      setTimeout(()=>{document.getElementById('loading').classList.add('hidden');render();},400);
    }).catch(e=>{document.getElementById('load-progress').textContent='⚠ '+e.message;});
}

}
